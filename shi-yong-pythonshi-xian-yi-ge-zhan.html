使用Python实现一个栈

实现基于LIFO(Last In First Out)即后进先出的栈

  • 主要方法有:
  • pop(): 从栈中删除最后一个元素并返回最后一个元素的值
  • push(): 往栈中添加一个元素
  • size(): 获取栈大小
  • is_empty(): 获取是否为空栈
#!/usr/bin/env python
# coding: utf-8

class MyStack(object):
    def __init__(self):
        self.items = []

    def pop(self):
        self.items.pop(-1)

    def push(self, x):
        self.items.append(x)

    def size(self):
        return len(self.items)

    def is_empty(self):
        return self.items == []

if __name__ == '__main__':
    my_stack = MyStack()
    print my_stack.size()
    my_stack.push(1)
    my_stack.pop()
    print my_stack.size()
    print my_stack.is_empty()
