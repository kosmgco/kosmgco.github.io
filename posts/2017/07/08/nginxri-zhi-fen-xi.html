Nginx日志分析

Nginx log

日志格式

log_format combined '$remote_addr - $remote_user [$time_local]'
                                   ' "$request" $status $body_bytes_sent'
                                   ' "$http_referer" "$http_user_agent"';
# $remote_addr 请求客户端地址
# $remote_user 请求客户端用户
# $time_local 当前机器的时间
# $request 请求头(请求方法，路径，协议)
# $status 返回状态码
# $body_byte_sent 返回数据长度
# $http_referer 来源
# $http_user_agent 请求工具

cat access.log | wc -l
