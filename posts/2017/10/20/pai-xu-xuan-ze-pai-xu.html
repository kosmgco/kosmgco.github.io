排序 - 选择排序

algorithms data-structure

选择排序的思想: 首先找到数组中最小的元素，然后将之与第一个元素交换位置。再找到剩余元素中最小的元素，与第二个元素交换位置。循环往复，即可完成选择排序。

选择排序的时间复杂度为O(n²)

最好的情况全部元素已经有序，则 交换次数为0；最差的情况，全部元素逆序，就要交换 n-1 次；

Python 实现

#!/usr/bin/env python
# coding: utf-8

class Selection(object):
    def __init__(self, l):
        self.l = l

    def sort(self):
        length = len(self.l)
        for i in range(0, length):
            m = i
            for j in range(i + 1, length):
                if self.less(self.l[j], self.l[m]):
                    m = j
            self.exchange(i, m)

    def less(self, a, b):
        return a < b

    def exchange(self, i, j):
        self.l[i], self.l[j] = self.l[j], self.l[i]

    def show(self):
        return ' '.join([str(item) for item in self.l])

    def is_sorted(self):
        for index in range(1, len(self.l))
            if self.less(self.l[index], self.l[index - 1]):
                return False
        return True


if __name__ == '__main__':
    a = Selection([1, 3, 2])
    a.sort()
    print a.show()

Golang 实现

package main

import (
    "fmt"
    "reflect"
)

type Selection []interface{}

func (s *Selection)sort() {
    length := len(*s)
    for i := 0; i < length; i ++{
        min := i
        for j := i + 1; j < length; j ++{
            if ok, err := s.less(j, min); err == nil && ok{
                min = j
            }
        }
        s.exchange(i, min)
    }
}

func (s *Selection) less(a, b int) (bool, error) {
    switch ((*s)[a]).(type) {
    case int:
        return (*s)[a].(int) < (*s)[b].(int), nil
    case string:
        return (*s)[a].(string) < (*s)[b].(string), nil
    case int64:
        return (*s)[a].(int64) < (*s)[b].(int64), nil
    default:
        return false, fmt.Errorf("Unsupported type: %s", reflect.TypeOf((*s)[a]))
    }
}

func (s *Selection) isSorted() bool {
    for index := 1; index < len(*s); index ++{
        if ok, err := s.less(index, index - 1); err == nil && ok{
            return false
        }
    }
    return true
}

func (s *Selection) exchange(i, j int) {
    (*s)[i], (*s)[j] = (*s)[j], (*s)[i]
}

func (s *Selection) show() {
    fmt.Println(*s)
}

func main() {
    s := Selection{1, 3, 2}
    s.sort()
    s.show()
}
