Golang JSON 自定义类型的序列化与反序列化

Golang json

The Go Playground

package main

import (
    "time"
    "fmt"
    "encoding/json"
    "strings"
)

type MyTime time.Time // 自定义类型

func (m MyTime) MarshalJSON() ([]byte, error) {
    // 实现 Marshaler 接口
    stamp := fmt.Sprintf("\"%s\"", time.Time(m).Format("2006-01-02 15:04:05"))
    return []byte(stamp), nil
}

func (m *MyTime) UnmarshalJSON(data []byte) error {
    // 实现 Unmarshaler 接口
    t, err := time.Parse("2006-01-02 15:04:05", strings.Trim(string(data), "\""))
    if err != nil {
        return err
    }
    *m = MyTime(t)
    return nil
}

type MyStr string

func (m MyStr) MarshalJSON() ([]byte, error) {
    return []byte("321321"), nil
}

func (m *MyStr) UnmarshalJSON(data []byte) error {
    *m = "12321321"
    return nil
}

type A struct {
    BBB MyTime `json:"bbb"`
    CCC string `json:"ccc"`
    DDD MyStr  `json:"ddd"`
}

func main() {
    a := &A{
        BBB: MyTime(time.Now()),
        CCC: "1",
        DDD: "ddd",
    }
    r, err := json.Marshal(a)
    if err != nil {
        fmt.Println(err)
    }
    fmt.Println(string(r))
    c := `{"bbb":"2018-01-16 14:43:10","ccc":"1","ddd":321321}`

    v := &A{}
    json.Unmarshal([]byte(c), &v)
    fmt.Println(time.Time(v.BBB), v.DDD)
}
// Output:
// {"bbb":"2009-11-10 23:00:00","ccc":"1","ddd":321321}
// 2018-01-16 14:43:10 +0000 UTC 12321321
