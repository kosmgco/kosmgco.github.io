定时更换ubuntu系统桌面背景 

#!/usr/bin/env python
# coding:utf-8
# bing.py

import re
import os
import requests
import datetime

req = requests.Session()

setting_to_background_cmd = 'gsettings set org.gnome.desktop.background draw-background false && gsettings set org.gnome.desktop.background picture-uri file:///home/scdev/images/{date}/{filename} && gsettings set org.gnome.desktop.background draw-background true'

req.headers = {
    'User-Agent': 'Mozilla/5.0 (X11; Linux x86_64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/53.0.2785.116 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8'
}
url = 'http://www.bing.com/HPImageArchive.aspx?format=js&idx=0&n=1&nc=1409879295618&pid=hp'

url = req.get(url).json().get('images', [])[0].get('url', '')

url = url if url.startswith('http') else 'http://www.bing.com' + url

base_path = os.path.abspath('')

if os.path.isdir(base_path + '/images/' + datetime.datetime.today().strftime('%Y/%m/%d/')):
    pass
else:
    os.system('mkdir -p ' + base_path + '/images/' + datetime.datetime.today().strftime('%Y/%m/%d/'))

with open(base_path + '/images/' + datetime.datetime.today().strftime('%Y/%m/%d/')  +url.split('/')[-1], 'wb') as buf:
    buf.write(req.get(url).content)

os.system(setting_to_background_cmd.format(
    date = datetime.datetime.today().strftime('%Y/%m/%d'),
    filename = url.split('/')[-1]
))

设置crontab定时任务就好了

0 10 * * * python bing.py 1>/dev/null 2>/dev/null &

这表示每天10点0分，执行bing.py这个脚本
