retrying模块的使用

install

pip install retrying

usage

from retrying import retry

@retry
def case1():
    pass


@retry(wait_random_min=1000, wait_random_max=2000)
def case2():
    pass


@retry(stop_max_attempt_number=3)
def case3():
    pass
